# C++用户定义字面量

```C++
#include <iostream>
#include <string>

std::string operator""_r1(size_t size)
{
    std::cout << "_r1" << std::endl;
    return std::to_string(size);
}

std::string operator""_r2(const char *str)
{
    std::cout << "_r2" << std::endl;
    return str;
}

int main(int argc, char *argv[])
{
    std::cout << 123456_r1 << std::endl;    // _r1

    std::cout << 123456_r2 << std::endl;    // _r2
    return 0;
}
```
